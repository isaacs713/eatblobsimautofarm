local normalSize = UDim2.new(0, 200, 0, 50)
local bigSize = UDim2.new(0, 240, 0, 60)
local closeNormalSize = UDim2.new(0, 28, 0, 28)
local closeBigSize = UDim2.new(0, 36, 0, 36)
local isTeleporting = false
local teleportTask = nil
button.MouseButton1Click:Connect(function()
    button.Size = bigSize
    task.spawn(function()
        task.wait(0.6)
        button.Size = normalSize
    end)
    if not isTeleporting then
        isTeleporting = true
        button.Text = "Stop Teleporting"
        teleportTask = task.spawn(function()
            local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
            local orbsFolder = Workspace:FindFirstChild("Orbs")
            if character and orbsFolder then
                local orbs = orbsFolder:GetChildren()
                local orbCount = #orbs
                local i = 1
                while isTeleporting and i <= orbCount do
                    local orb = orbs[i]
                    if orb:IsA("BasePart") then
                        character:PivotTo(orb:GetPivot())
                        task.wait(0.1)
                    end
                    i = i + 1
                end
            end
            isTeleporting = false
            button.Text = "Teleport To Orbs"
        end)
    else
        isTeleporting = false
        button.Text = "Teleport To Orbs"
    end
end)

local function showConfirmFrame()
    confirmFrame.Visible = true
    confirmFrame.Position = UDim2.new(0.5, 0, 1, -50)
    confirmFrame.AnchorPoint = Vector2.new(0.5, 0.5)
    local viewportSize = mainGui.AbsoluteSize
    local centerX = viewportSize.X/2
    local centerY = viewportSize.Y/2
    local tweenInfo = TweenInfo.new(0.4, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    local tween = TweenService:Create(confirmFrame, tweenInfo, {Position = UDim2.new(0.5, 0, 0.5, 0)})
    tween:Play()
end

closeBtn.MouseButton1Click:Connect(function()
    local tweenInfo = TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    local growTween = TweenService:Create(closeBtn, tweenInfo, {Size = closeBigSize})
    growTween:Play()
    growTween.Completed:Connect(function()
        local shrinkTween = TweenService:Create(closeBtn, tweenInfo, {Size = closeNormalSize})
        shrinkTween:Play()
    end)
    mainFrame.Visible = false
    showConfirmFrame()
end)
yesBtn.MouseButton1Click:Connect(function()
    confirmFrame.Visible = false
    mainFrame.Visible = false
    if mainGui then
        mainGui:Destroy()
    end
end)

noBtn.MouseButton1Click:Connect(function()

    local tween = TweenService:Create(confirmFrame, tweenInfo, {Position = UDim2.new(0.5, 0, 1, -50)})
    tween:Play()
    tween.Completed:Connect(function()
        confirmFrame.Visible = false
        mainFrame.Visible = true
        confirmFrame.Position = UDim2.new(0.5, 0, 1, -50)
    end)
end)
